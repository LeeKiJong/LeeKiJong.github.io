<h3>알고스팟</h3>
<h1>1261번 문제</h1>

링크 : [https://www.acmicpc.net/problem/1261](https://www.acmicpc.net/problem/1261)  
```cpp
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <deque>
#define MAX 100
using namespace std;

int py[4] = { -1, 0, 1,0 };		
int px[4] = { 0, -1, 0, 1 };
int wall[MAX][MAX];
int visited[MAX][MAX];
int M, N;

struct Point {
	int y, x;
};
bool is_inside(int ny, int nx) {
	return (ny >= 0 && ny < N && nx >= 0 && nx < M);
}

int bfs() {
	deque<Point> dq;
	dq.push_back({ 0, 0 });
	visited[0][0] = 1;
	while (!dq.empty()) {
		int y = dq.front().y;
		int x = dq.front().x;
		dq.pop_front();

		if (y == N-1 && x == M-1) return visited[N-1][M-1] - 1;

		for (int i = 0; i < 4; i++) {
			int ny = y + py[i];
			int nx = x + px[i];

			//맵을 벗어나지 않고, 갈 수 있는 길이고, 방문하지 않았다면
			if (is_inside(ny, nx) && wall[ny][nx] == 0 && visited[ny][nx] == 0) {
				dq.push_front({ ny, nx });
				visited[ny][nx] = visited[y][x];
			}
			//맵을 벗어나지 않고, 갈 수 없는 길이고, 방문하지 않았다면
			if (is_inside(ny, nx) && wall[ny][nx] == 1 && visited[ny][nx] == 0) {
				dq.push_back({ ny, nx });
				visited[ny][nx] = visited[y][x] +1;
			}
		}
	}

	return 0;

}
int main() {

	cin >> M >> N;

	for (int i = 0; i < N; i++) {
		for (int j = 0; j < M; j++) {
			scanf("%1d", &wall[i][j]);

		}
	}
	printf("%d\n", bfs());
	return 0;
}
```
