<h3>톱니바퀴</h3>
<h1>14891</h1>
링크 : [https://www.acmicpc.net/problem/14891](https://www.acmicpc.net/problem/14891)  

```cpp
#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <cmath>
using namespace std;

int Gear[4][8];
int K, num, d;

void rotate(int num, int d) {
	int a[8];     //데이터 보존을 위해 따로 배열 변수 선언
	if (d == 1) { //시계
		for (int i = 0; i < 7; i++) {
			a[i + 1] = Gear[num][i];
		}
		a[0] = Gear[num][7];
	}else { //반시계
		for (int i = 7; i > 0; i--) {
			a[i-1] = Gear[num][i];
		}
		a[7] = Gear[num][0];
	}
	for (int i = 0; i < 8; i++) {
		Gear[num][i] = a[i];      //회전이 완료되면 기존 배열에 삽입
	}
}

int main(void) {
	
	for (int i = 0; i < 4; i++) {
		for (int j = 0; j < 8; j++) {
			scanf("%1d", &Gear[i][j]);      //톱니바퀴의 개수(4) * 톱니의 개수(8) = 32, Gear[0][0] ---> Gear[3][7]
		}
	}
	scanf("%d", &K);

	while(K--) {
		scanf("%d %d", &num, &d);
		num--;
	
		//1시계, -1반시계
		int cur[4] = {0};       //톱니바퀴가 최종적으로 몇 번 회전을 해야 하는지 알기 위함
		cur[num] = d;
		
		for (int i = num; i < 3; i++) {      //첫 번째 톱니바퀴면 오른쪽으로 톱니가 맞닿는 3부분 검사
			if (Gear[i][2] != Gear[i + 1][6]) cur[i + 1] = -cur[i]; 
		}
		
		for (int i = num; i > 0; i--) {
			if (Gear[i][6] != Gear[i - 1][2]) cur[i - 1] = -cur[i];
		}

		for (int i = 0; i < 4; i++) {
			if (cur[i]) {
				rotate(i, cur[i]);
			}
			
		}
	}
		
	//점수
	int score = 0;
	for (int i = 0; i < 4; i++) {
		if (Gear[i][0]) score += pow(2, i);
	}
	cout << score << endl;

	return 0;
}
```
