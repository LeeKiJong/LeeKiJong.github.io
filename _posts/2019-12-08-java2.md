import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		int H, M;
		H = scanner.nextInt();
		M = scanner.nextInt();
		if((M-45)<0) {
			M=M+60-45;
			if(H==0)
				H=H+24-1;
			else
				H=H-1;
		}
		else
			M=M-45;
		
		System.out.println(H + " " + M);
	}	

}

import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner scanner = new Scanner(System.in);
		int A, B, C;
		A = scanner.nextInt();
		B = scanner.nextInt();
		C = scanner.nextInt();
		if(A>=B)
			if(A<=C)
				System.out.println(A);
			else
				if(B>=C)
					System.out.println(B);
				else
					System.out.println(C);
		else
			if(B>=C)
				if(A>=C)
					System.out.println(A);
				else
					System.out.println(C);
			else
				System.out.println(B);
	}	
}
